########################################
#
# CMake file for Python wrapper:_Maths
#
########################################

# Third Party dependencies of this module
include_directories( ${BOOST_INCLUDE_DIR}
                     ${PYTHON_INCLUDE_DIR}
                     ${SIRE_GSL_INCLUDE_PATH} )

# Sire include paths
include_directories( BEFORE ${SIRE_INCLUDE_DIR}/SireMaths
                            ${SIRE_INCLUDE_DIR} )

# Other python wrapping directories
include_directories(${CMAKE_SOURCE_DIR})

# Need higher arity as have 16 arguments for a Matrix4
add_definitions( "-DBOOST_PYTHON_MAX_ARITY=17" )

# Get the list of autogenerated files
include(CMakeAutogenFile.txt)

# Define the sources in SireBase
set ( WRAPMATHS_SOURCES

      _Maths.main.cpp
      _Maths_global_variables.pyman.cpp

      ${PYPP_SOURCES}

    )

# Create the library that holds all of the class wrappers
add_library (Maths ${WRAPMATHS_SOURCES})

set_target_properties (Maths
                       PROPERTIES
                         VERSION ${SIRE_VERSION}
                         SOVERSION ${SIRE_VERSION_MAJOR}
                         OUTPUT_NAME "Maths"
                         PREFIX "_"
                         SUFFIX ${PYTHON_MODULE_EXTENSION}
		       )

target_link_libraries (Maths
                       SirePython
                       SIRE_SireMaths
                       ${Boost_LIBRARIES}
                      )

include( LimitSirePythonExportSymbols )
export_this_symbol_only( "init_Maths" "_init_Maths" )

# installation
set( INSTALLDIR ${SIRE_PYTHON}/sire/legacy/Maths )

install( TARGETS Maths
         LIBRARY DESTINATION ${INSTALLDIR}
         ARCHIVE DESTINATION ${INSTALLDIR}
         RUNTIME DESTINATION ${INSTALLDIR}
        )

install( FILES __init__.py
         DESTINATION ${INSTALLDIR}
       )
