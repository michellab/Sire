########################################
#
# CMake file for Python wrapper:_Move
#
########################################

# Third Party dependencies of this module
include_directories( ${BOOST_INCLUDE_DIR}
                     ${PYTHON_INCLUDE_DIR}
                     ${SIRE_GSL_INCLUDE_PATH} )

# Sire include paths
include_directories( BEFORE ${SIRE_INCLUDE_DIR}/SireMove
                            ${SIRE_INCLUDE_DIR} )

if (SIRE_USE_OPENMM)
  if (${SIRE_OpenMM_FOUND})
      include_directories( ${SIRE_OpenMM_INCLUDE_DIR} )
      add_definitions( "-DSIRE_USE_OPENMM" )
      set( SIRE_OPENMM_LIBRARIES ${SIRE_OpenMM_LIBRARIES} )
      include(CMakeOpenMMFile.txt)
  else()
      include(CMakeNoOpenMMFile.txt)
  endif()
else()
  include(CMakeNoOpenMMFile.txt)
endif()

# Other python wrapping directories
include_directories(${CMAKE_SOURCE_DIR})

# Get the list of autogenerated files
include(CMakeAutogenFile.txt)

foreach(FILE ${PYPP_OPENMM_SOURCES})
  get_filename_component(BASEFILE ${FILE} NAME)
  list(FIND PYPP_SOURCES ${BASEFILE} FILEINDEX)
  if (FILEINDEX GREATER -1)
    list(REMOVE_AT PYPP_SOURCES ${FILEINDEX})
  endif()
endforeach()

# Define the sources in Move
set ( WRAPMOVE_SOURCES

      _Move.main.cpp

      ${PYPP_SOURCES}

      ${PYPP_OPENMM_SOURCES}
    )

# Create the library that holds all of the class wrappers
add_library (Move ${WRAPMOVE_SOURCES})

set_target_properties (Move
                       PROPERTIES
                         VERSION ${SIRE_VERSION}
                         SOVERSION ${SIRE_VERSION_MAJOR}
                         OUTPUT_NAME "Move"
                         PREFIX "_"
                         SUFFIX ${PYTHON_MODULE_EXTENSION}
                        )

target_link_libraries (Move
                       SirePython
                       SIRE_SireMove
                       ${Boost_LIBRARIES}
                       ${SIRE_OPENMM_LIBRARIES}
                      )

include( LimitSirePythonExportSymbols )
export_this_symbol_only( "init_Move" "_init_Move" )

# installation
set( INSTALLDIR ${SIRE_PYTHON}/sire/legacy/Move )

install( TARGETS Move
         LIBRARY DESTINATION ${INSTALLDIR}
         ARCHIVE DESTINATION ${INSTALLDIR}
         RUNTIME DESTINATION ${INSTALLDIR}
        )

install( FILES __init__.py
         DESTINATION ${INSTALLDIR}
       )
